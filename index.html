6) Finding Mouse coordinates using python

        import pyautogui as pg
        print('c+c to stop')
        try:
            while True:
                x,y=pg.position()
                coordinates='X: '+str(x).ljust(4)+'Y: '+str(y).ljust(4)
                print(coordinates,end='')
                print('\b' * len(coordinates),end='',flush=True)
        
        except KeyboardInterrupt:
